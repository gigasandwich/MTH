    \frac{dy}{y} &= \frac{-3x}{x^2+1} dx \\
    \int \frac{1}{y} \, dy &= \int -\frac{3x}{x^2+1} \, dx \\

    y &= \lambda (x^2 + 1)^{-3/2}, \quad \lambda = e^C \quad (1)
\end{align*}
